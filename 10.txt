#include <LPC214x.h> 
Void UART0_Init(void) { 
VPBDIV = 0x01; // PCLK = CCLK (12 MHz) 
PINSEL0 = 0x00000005;  
U0LCR = 0x83;  
U0DLL = 78;  
U0DLM = 0; 
U0LCR = 0x03; } 
Void UART0_Send(char ch) { 
While(!(U0LSR & 0x20)); 
U0THR = ch; } 
Int main(void) { 
Int I;  
UART0_Init(); 
While(1) { 
UART0_Send(‘O’); 
UART0_Send(‘K’); 
UART0_Send(‘\r’); 
UART0_Send(‘\n’); 
For(i=0; i<5000000; i++);  } }
