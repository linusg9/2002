#include <LPC214x.h> 
void UART0_Init(void); 
void UART0_TxChar(char ch); 
void UART0_TxString(char *str); 
int main() { 
UART0_Init();  
UART0_TxString("Hello World\r\n"); 
while(1); } 
void UART0_Init(void) { 
PINSEL0 |= 0x00000005;  
U0LCR = 0x83; // 8-bit, 1 stop bit, DLAB=1 
U0DLL = 78;  
U0DLM = 0; 
U0LCR = 0x03; // DLAB=0 (lock divisor) } 
void UART0_TxChar(char ch) { 
while(!(U0LSR & 0x20));  
U0THR = ch;  } 
void UART0_TxString(char *str) { 
while(*str) { 
UART0_TxChar(*str); 
str++; } }
